    Javascript

      
  <script>
    const hbg = document.querySelectorAll('.hamburger-btn');
    hbg.addEventListener('click', () => {
      if (hbg.classList.contains('active')) hbg.classList.remove('active');
      else hbg.classList.add('active');
    });
  </script>
